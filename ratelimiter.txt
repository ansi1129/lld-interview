🧠 6. Algorithms (VERY IMPORTANT)
🔥 Fixed Window
1. Maintain counter per window
2. If count < limit → allow
3. Else → reject
4. Reset after window expires
🔥 Token Bucket
1. Tokens refill at fixed rate
2. Each request consumes 1 token
3. If token available → allow
4. Else → reject
💻 7. COMPLETE IMPLEMENTATION (INTERVIEW READY)
package main

import (
	"fmt"
	"sync"
	"time"
)

//////////////////////////////////////////////////////
// INTERFACE
//////////////////////////////////////////////////////

type RateLimiter interface {
	Allow(key string) bool
}

//////////////////////////////////////////////////////
// FIXED WINDOW
//////////////////////////////////////////////////////

type FixedWindowLimiter struct {
	limit      int
	windowSize time.Duration
	store      map[string]int
	timestamp  map[string]time.Time
	mutex      sync.Mutex
}

func NewFixedWindowLimiter(limit int, window int) *FixedWindowLimiter {
	return &FixedWindowLimiter{
		limit:      limit,
		windowSize: time.Duration(window) * time.Second,
		store:      make(map[string]int),
		timestamp:  make(map[string]time.Time),
	}
}

func (f *FixedWindowLimiter) Allow(key string) bool {
	f.mutex.Lock()
	defer f.mutex.Unlock()

	now := time.Now()

	if t, ok := f.timestamp[key]; !ok || now.Sub(t) > f.windowSize {
		f.timestamp[key] = now
		f.store[key] = 1
		return true
	}

	if f.store[key] < f.limit {
		f.store[key]++
		return true
	}

	return false
}

//////////////////////////////////////////////////////
// TOKEN BUCKET
//////////////////////////////////////////////////////

type TokenBucket struct {
	capacity   int
	tokens     map[string]int
	refillRate int
	lastRefill map[string]time.Time
	mutex      sync.Mutex
}

func NewTokenBucket(capacity int, refill int) *TokenBucket {
	return &TokenBucket{
		capacity:   capacity,
		refillRate: refill,
		tokens:     make(map[string]int),
		lastRefill: make(map[string]time.Time),
	}
}

func (t *TokenBucket) Allow(key string) bool {
	t.mutex.Lock()
	defer t.mutex.Unlock()

	now := time.Now()

	last, ok := t.lastRefill[key]
	if !ok {
		t.tokens[key] = t.capacity
		t.lastRefill[key] = now
	}

	elapsed := int(now.Sub(last).Seconds())
	refill := elapsed * t.refillRate

	if refill > 0 {
		t.tokens[key] = min(t.capacity, t.tokens[key]+refill)
		t.lastRefill[key] = now
	}

	if t.tokens[key] > 0 {
		t.tokens[key]--
		return true
	}

	return false
}

//////////////////////////////////////////////////////
// FACTORY
//////////////////////////////////////////////////////

func GetLimiter(algo string) RateLimiter {
	if algo == "FIXED" {
		return NewFixedWindowLimiter(5, 10)
	}
	return NewTokenBucket(5, 1)
}

//////////////////////////////////////////////////////
// MAIN
//////////////////////////////////////////////////////

func main() {

	limiter := GetLimiter("FIXED")

	for i := 0; i < 10; i++ {
		if limiter.Allow("user1") {
			fmt.Println("Allowed")
		} else {
			fmt.Println("Blocked")
		}
	}
}

func min(a, b int) int {
	if a < b {
		return a
	}
	return b
}
